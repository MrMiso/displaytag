22  package org.displaytag.jsptests;
23  
24  import org.displaytag.test.DisplaytagCase;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  import com.meterware.httpunit.GetMethodWebRequest;
29  import com.meterware.httpunit.WebRequest;
30  import com.meterware.httpunit.WebResponse;
31  import com.meterware.httpunit.WebTable;
32  
33  /**
34   * Tests for the htmlId attribute.
35   *
36   * @author Fabrizio Giustina
37   *
38   * @version $Revision$ ($Author$)
39   */
40  public class HtmlIdTest extends DisplaytagCase {
41  
42      /**
43       * Gets the jsp name.
44       *
45       * @return the jsp name
46       *
47       * @see org.displaytag.test.DisplaytagCase#getJspName()
48       */
49      @Override
50      public String getJspName() {
51          return "htmlid.jsp";
52      }
53  
54      /**
55       * Check content and ids in generated tables.
56       *
57       * @throws Exception
58       *             any axception thrown during test.
59       */
60      @Override
61      @Test
62      public void doTest() throws Exception {
63          final WebRequest request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
64          WebResponse response;
65  
66          response = this.runner.getResponse(request);
67  
68          final WebTable[] tables = response.getTables();
69          Assert.assertEquals("Wrong number of tables in result.", 3, tables.length);
70  
71          for (int j = 0; j < tables.length; j++) {
72              Assert.assertEquals("invalid id", "html" + (j + 1), tables[j].getID());
73              Assert.assertEquals("Unexpected value in table cell", "bee", tables[j].getCellAsText(1, 0));
74          }
75  
76      }
77  }

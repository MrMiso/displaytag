22  package org.displaytag.jsptests;
23  
24  import org.displaytag.properties.MediaTypeEnum;
25  import org.displaytag.tags.TableTagParameters;
26  import org.displaytag.test.DisplaytagCase;
27  import org.displaytag.util.ParamEncoder;
28  import org.junit.jupiter.api.Assertions;
29  import org.junit.jupiter.api.Test;
30  
31  import com.meterware.httpunit.GetMethodWebRequest;
32  import com.meterware.httpunit.HttpInternalErrorException;
33  import com.meterware.httpunit.WebRequest;
34  import com.meterware.httpunit.WebResponse;
35  
36  /**
37   * Tests the ResponseOverrideFilter.
38   *
39   * @author Fabrizio Giustina
40   *
41   * @version $Revision$ ($Author$)
42   */
43  class FilterTest extends DisplaytagCase {
44  
45      /**
46       * Gets the jsp name.
47       *
48       * @return the jsp name
49       *
50       * @see org.displaytag.test.DisplaytagCase#getJspName()
51       */
52      @Override
53      public String getJspName() {
54          return "filter.jsp";
55      }
56  
57      /**
58       * Do test.
59       *
60       * @throws Exception
61       *             the exception
62       *
63       * @see org.displaytag.test.DisplaytagCase#doTest()
64       */
65      @Override
66      @Test
67      public void doTest() throws Exception {
68          final ParamEncoder encoder = new ParamEncoder("table");
69          final String mediaParameter = encoder.encodeParameterName(TableTagParameters.PARAMETER_EXPORTTYPE);
70  
71          WebRequest request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
72          request.setParameter(mediaParameter, Integer.toString(MediaTypeEnum.XML.getCode()));
73  
74          try {
75              // check if page need a filter (unfiltered request)
76              this.runner.getResponse(request);
77              Assertions.fail("Request works also without a filter. Can't test it properly.");
78          } catch (final HttpInternalErrorException e) {
79              // it's ok
80          }
81  
82          request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
83          request.setParameter(mediaParameter, Integer.toString(MediaTypeEnum.XML.getCode()));
84  
85          // this enable the filter!
86          request.setParameter(TableTagParameters.PARAMETER_EXPORTING, "1");
87  
88          final WebResponse response = this.runner.getResponse(request);
89  
90          Assertions.assertEquals("text/xml", response.getContentType(), "Expected a different content type.");
91      }
92  
93  }

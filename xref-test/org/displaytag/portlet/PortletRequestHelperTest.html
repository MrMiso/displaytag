100     /**
101      * Test request parameters.
102      */
103     @Test
104     void testRequestParameters() {
105         final MockPageContext pageContext = new MockPageContext();
106         final MockPortletRequest request = new MockPortletRequest();
107 
108         request.setParameter("STRING_PARAM", "STRING_VALUE");
109         request.setParameter("INTEGER_PARAM", "31337");
110 
111         pageContext.setAttribute(PortletRequestHelper.JAVAX_PORTLET_REQUEST, request);
112         pageContext.setAttribute(PortletRequestHelper.JAVAX_PORTLET_RESPONSE, new MockRenderResponse());
113 
114         final PortletRequestHelper helper = new PortletRequestHelper(pageContext);
115 
116         final String strVal = helper.getParameter("STRING_PARAM");
117         Assertions.assertEquals("STRING_VALUE", strVal);
118 
119         final Integer intVal = helper.getIntParameter("INTEGER_PARAM");
120         Assertions.assertEquals(Integer.valueOf(31337), intVal);
121 
122         final Integer nullIntVal = helper.getIntParameter("STRING_PARAM");
123         Assertions.assertNull(nullIntVal);
124 
125         final Map<String, String[]> params = helper.getParameterMap();
126         Assertions.assertEquals(2, params.size());
127 
128         final String[] expextedStrArryVal = { "STRING_VALUE" };
129         final String[] strArryVal = params.get("STRING_PARAM");
130         Assertions.assertEquals(expextedStrArryVal.length, strArryVal.length);
131         Assertions.assertEquals(expextedStrArryVal[0], strArryVal[0]);
132 
133         final String[] expextedIntArryVal = { "31337" };
134         final String[] intArryVal = params.get("INTEGER_PARAM");
135         Assertions.assertEquals(expextedIntArryVal.length, intArryVal.length);
136         Assertions.assertEquals(expextedIntArryVal[0], intArryVal[0]);
137     }

37  /**
38   * Mock implementation of the {@link javax.portlet.PortletURL} interface.
39   *
40   * @author John A. Lewis
41   * @author Juergen Hoeller
42   * @since 2.0
43   */
44  public class MockPortletURL extends MockBaseURL implements PortletURL, RenderURL, ActionURL {
45  
46  	/** The Constant URL_TYPE_RENDER. */
47  	public static final String URL_TYPE_RENDER = "render";
48  
49  	/** The Constant URL_TYPE_ACTION. */
50  	public static final String URL_TYPE_ACTION = "action";
51  
52  
53  	/** The portal context. */
54  	private final PortalContext portalContext;
55  
56  	/** The url type. */
57  	private final String urlType;
58  
59  	/** The window state. */
60  	private WindowState windowState;
61  
62  	/** The portlet mode. */
63  	private PortletMode portletMode;
64  
65  
66  	/**
67  	 * Create a new MockPortletURL for the given URL type.
68  	 * @param portalContext the PortalContext defining the supported
69  	 * PortletModes and WindowStates
70  	 * @param urlType the URL type, for example "render" or "action"
71  	 * @see #URL_TYPE_RENDER
72  	 * @see #URL_TYPE_ACTION
73  	 */
74  	public MockPortletURL(PortalContext portalContext, String urlType) {
75  		Assertions.assertNotNull(portalContext, "PortalContext is required");
76  		this.portalContext = portalContext;
77  		this.urlType = urlType;
78  	}
79  
80  
81  	//---------------------------------------------------------------------
82  	// PortletURL methods
83  	//---------------------------------------------------------------------
84  
85  	@Override
86  	public void setWindowState(WindowState windowState) throws WindowStateException {
87  		if (!CollectionUtils.contains(this.portalContext.getSupportedWindowStates(), windowState)) {
88  			throw new WindowStateException("WindowState not supported", windowState);
89  		}
90  		this.windowState = windowState;
91  	}
92  
93  	@Override
94  	public WindowState getWindowState() {
95  		return this.windowState;
96  	}
97  
98  	@Override
99  	public void setPortletMode(PortletMode portletMode) throws PortletModeException {
100 		if (!CollectionUtils.contains(this.portalContext.getSupportedPortletModes(), portletMode)) {
101 			throw new PortletModeException("PortletMode not supported", portletMode);
102 		}
103 		this.portletMode = portletMode;
104 	}
105 
106 	@Override
107 	public PortletMode getPortletMode() {
108 		return this.portletMode;
109 	}
110 
111 	@Override
112 	public void removePublicRenderParameter(String name) {
113 		this.parameters.remove(name);
114 	}
115 
116 
117 	@Override
118 	public String toString() {
119 		StringBuilder sb = new StringBuilder();
120 		sb.append(encodeParameter("urlType", this.urlType));
121 		if (this.windowState != null) {
122 			sb.append(";").append(encodeParameter("windowState", this.windowState.toString()));
123 		}
124 		if (this.portletMode != null) {
125 			sb.append(";").append(encodeParameter("portletMode", this.portletMode.toString()));
126 		}
127 		for (Map.Entry<String, String[]> entry : this.parameters.entrySet()) {
128 			sb.append(";").append(encodeParameter("param_" + entry.getKey(), entry.getValue()));
129 		}
130 		return (isSecure() ? "https:" : "http:") +
131 				"//localhost/mockportlet?" + sb.toString();
132 	}

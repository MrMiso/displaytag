22  package org.displaytag.export;
23  
24  import java.io.IOException;
25  import java.io.OutputStream;
26  
27  import com.itextpdf.text.Document;
28  import com.itextpdf.text.DocumentException;
29  import com.itextpdf.text.pdf.BaseFont;
30  import com.itextpdf.text.pdf.PdfContentByte;
31  import com.itextpdf.text.pdf.PdfPTable;
32  import com.itextpdf.text.pdf.PdfPageEventHelper;
33  import com.itextpdf.text.pdf.PdfTemplate;
34  import com.itextpdf.text.pdf.PdfWriter;
35  
36  /**
37   * PDF exporter using iText.
38   *
39   * @author Jorge L. Barroso
40   *
41   * @version $Revision$ ($Author$)
42   */
43  public class DefaultPdfExportView extends DefaultItextExportView {
44  
45      /**
46       * Gets the mime type.
47       *
48       * @return "application/pdf"
49       *
50       * @see org.displaytag.export.BaseExportView#getMimeType()
51       */
52      @Override
53      public String getMimeType() {
54          return "application/pdf"; //$NON-NLS-1$
55      }
56  
57      /**
58       * Initializes the PDF writer this export view uses to write the table document.
59       *
60       * @param document
61       *            The iText document to be written.
62       * @param out
63       *            The output stream to which the document is written.
64       *
65       * @throws DocumentException
66       *             If something goes wrong during initialization.
67       */
68      @Override
69      protected void initItextWriter(final Document document, final OutputStream out) throws DocumentException {
70          PdfWriter.getInstance(document, out).setPageEvent(new PageNumber());
71      }
72  
73      /**
74       * Prints a page number at the bottom of each page. Based on
75       * http://itextdocs.lowagie.com/examples/com/lowagie/examples/directcontent/pageevents/PageNumbersWatermark.java
76       *
77       * @author Jorge L. Barroso
78       *
79       * @version $Revision$ ($Author$)
80       */
81      private static class PageNumber extends PdfPageEventHelper {
82  
83          /**
84           * On end page.
85           *
86           * @param writer
87           *            the writer
88           * @param document
89           *            the document
90           *
91           * @see com.lowagie.text.pdf.PdfPageEventHelper#onEndPage(com.lowagie.text.pdf.PdfWriter,
92           *      com.lowagie.text.Document)
93           */
94          @Override
95          public void onEndPage(final PdfWriter writer, final Document document) {
96              /** The headertable. */
97              final PdfPTable table = new PdfPTable(2);
98              /** A template that will hold the total number of pages. */
99              final PdfTemplate tpl = writer.getDirectContent().createTemplate(100, 100);
100             /** The font that will be used. */
101             BaseFont helv = null;
102             try {
103                 helv = BaseFont.createFont("Helvetica", BaseFont.WINANSI, false);
104             } catch (DocumentException | IOException e) {
105             }
106             final PdfContentByte cb = writer.getDirectContent();
107             // write the headertable
108             table.setTotalWidth(document.right() - document.left());
109             table.writeSelectedRows(0, -1, document.left(), document.getPageSize().getHeight() - 50, cb);
110             // compose the footer
111             final String text = "Page " + writer.getPageNumber();
112             final float textSize = helv.getWidthPoint(text, 12);
113             final float textBase = document.bottom() - 20;
114             cb.beginText();
115             cb.setFontAndSize(helv, 12);
116             final float adjust = helv.getWidthPoint("0", 12);
117             cb.setTextMatrix(document.right() - textSize - adjust, textBase);
118             cb.showText(text);
119             cb.endText();
120             cb.addTemplate(tpl, document.right() - adjust, textBase);
121         }
122     }
123 }

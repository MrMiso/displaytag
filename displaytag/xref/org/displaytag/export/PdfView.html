55  /**
56   * PDF exporter using IText. This class is provided more as an example than as a "production ready" class: users
57   * probably will need to write a custom export class with a specific layout.
58   *
59   * @author Ivan Markov
60   * @author Fabrizio Giustina
61   *
62   * @version $Revision$ ($Author$)
63   */
64  public class PdfView implements BinaryExportView {
65  
66      /**
67       * TableModel to render.
68       */
69      private TableModel model;
70  
71      /** export full list?. */
72      private boolean exportFull;
73  
74      /** include header in export?. */
75      private boolean header;
76  
77      /** decorate export?. */
78      private boolean decorated;
79  
80      /**
81       * This is the table, added as an Element to the PDF document. It contains all the data, needed to represent the
82       * visible table into the PDF
83       */
84      private PdfPTable tablePDF;
85  
86      /**
87       * The default font used in the document.
88       */
89      private Font smallFont;
90  
91      /**
92       * Sets the parameters.
93       *
94       * @param tableModel
95       *            the table model
96       * @param exportFullList
97       *            the export full list
98       * @param includeHeader
99       *            the include header
100      * @param decorateValues
101      *            the decorate values
102      *
103      * @see org.displaytag.export.ExportView#setParameters(TableModel, boolean, boolean, boolean)
104      */
105     @Override
106     public void setParameters(final TableModel tableModel, final boolean exportFullList, final boolean includeHeader,
107             final boolean decorateValues) {
108         this.model = tableModel;
109         this.exportFull = exportFullList;
110         this.header = includeHeader;
111         this.decorated = decorateValues;
112     }
113 
114     /**
115      * Initialize the main info holder table.
116      */
117     protected void initTable() {
118         this.tablePDF = new PdfPTable(this.model.getNumberOfColumns());
119         this.tablePDF.getDefaultCell().setVerticalAlignment(Element.ALIGN_TOP);
120         this.tablePDF.setWidthPercentage(100);
121 
122         this.smallFont = FontFactory.getFont(FontFactory.HELVETICA, 7, Font.NORMAL, new BaseColor(0, 0, 0));
123 
124     }
125 
126     /**
127      * Gets the mime type.
128      *
129      * @return "application/pdf"
130      *
131      * @see org.displaytag.export.BaseExportView#getMimeType()
132      */
133     @Override
134     public String getMimeType() {
135         return "application/pdf"; //$NON-NLS-1$
136     }
137 
138     /**
139      * The overall PDF table generator.
140      *
141      * @throws JspException
142      *             for errors during value retrieving from the table model
143      */
144     protected void generatePDFTable() throws JspException {
145         if (this.header) {
146             this.generateHeaders();
147         }
148         this.generateRows();
149     }
150 
151     /**
152      * Do export.
153      *
154      * @param out
155      *            the out
156      *
157      * @throws JspException
158      *             the jsp exception
159      *
160      * @see org.displaytag.export.BinaryExportView#doExport(OutputStream)
161      */
162     @Override
163     public void doExport(final OutputStream out) throws JspException {
164         try {
165             // Initialize the table with the appropriate number of columns
166             this.initTable();
167 
168             // Initialize the Document and register it with PdfWriter listener and the OutputStream
169             final Document document = new Document(PageSize.A4.rotate(), 60, 60, 40, 40);
170             document.addCreationDate();
171 
172             final PdfWriter writer = PdfWriter.getInstance(document, out);
173             writer.setPageEvent(new PdfPageEventHelper() {
174 
175                 @Override
176                 public void onEndPage(final PdfWriter writer, final Document document) {
177 
178                     ColumnText.showTextAligned(writer.getDirectContent(), Element.ALIGN_CENTER,
179                             new Phrase(TagConstants.EMPTY_STRING, PdfView.this.smallFont),
180                             (document.left() + document.right()) / 2, document.bottom() - 18, 0);
181                 }
182 
183             });
184 
185             // Fill the virtual PDF table with the necessary data
186             this.generatePDFTable();
187 
188             document.open();
189             document.add(this.tablePDF);
190             document.close();
191 
192         } catch (final Exception e) {
193             throw new PdfGenerationException(e);
194         }
195     }
196 
197     /**
198      * Generates the header cells, which persist on every page of the PDF document.
199      */
200     protected void generateHeaders() {
201         final Iterator<HeaderCell> iterator = this.model.getHeaderCellList().iterator();
202 
203         while (iterator.hasNext()) {
204             final HeaderCell headerCell = iterator.next();
205 
206             String columnHeader = headerCell.getTitle();
207 
208             if (columnHeader == null) {
209                 columnHeader = StringUtils.capitalize(headerCell.getBeanPropertyName());
210             }
211 
212             final PdfPCell hdrCell = this.getCell(columnHeader);
213             hdrCell.setGrayFill(0.9f);
214             this.tablePDF.addCell(hdrCell);
215 
216         }
217         this.tablePDF.setHeaderRows(1);
218     }
219 
220     /**
221      * Generates all the row cells.
222      *
223      * @throws JspException
224      *             for errors during value retrieving from the table model
225      */
226     protected void generateRows() throws JspException {
227         // get the correct iterator (full or partial list according to the exportFull field)
228         final RowIterator rowIterator = this.model.getRowIterator(this.exportFull);
229         // iterator on rows
230         while (rowIterator.hasNext()) {
231             final Row row = rowIterator.next();
232 
233             // iterator on columns
234             final ColumnIterator columnIterator = row.getColumnIterator(this.model.getHeaderCellList());
235 
236             while (columnIterator.hasNext()) {
237                 final Column column = columnIterator.nextColumn();
238 
239                 // Get the value to be displayed for the column
240                 final Object value = column.getValue(this.decorated);
241 
242                 final PdfPCell cell = this.getCell(value != null ? value.toString() : StringUtils.EMPTY);
243                 this.tablePDF.addCell(cell);
244             }
245         }
246     }
247 
248     /**
249      * Returns a formatted cell for the given value.
250      *
251      * @param value
252      *            cell value
253      *
254      * @return Cell
255      */
256     private PdfPCell getCell(final String value) {
257         final PdfPCell cell = new PdfPCell(new Phrase(new Chunk(StringUtils.trimToEmpty(value), this.smallFont)));
258         cell.setVerticalAlignment(Element.ALIGN_TOP);
259         cell.setLeading(8, 0);
260         cell.setPadding(2);
261         return cell;
262     }

266      *
267      * @author Fabrizio Giustina
268      *
269      * @version $Revision$ ($Author$)
270      */
271     static class PdfGenerationException extends BaseNestableJspTagException {
272 
273         /**
274          * D1597A17A6.
275          */
276         private static final long serialVersionUID = 899149338534L;
277 
278         /**
279          * Instantiate a new PdfGenerationException with a fixed message and the given cause.
280          *
281          * @param cause
282          *            Previous exception
283          */
284         public PdfGenerationException(final Throwable cause) {
285             super(PdfView.class, Messages.getString("PdfView.errorexporting"), cause); //$NON-NLS-1$
286         }
287 
288         /**
289          * Gets the severity.
290          *
291          * @return the severity
292          *
293          * @see org.displaytag.exception.BaseNestableJspTagException#getSeverity()
294          */
295         @Override
296         public SeverityEnum getSeverity() {
297             return SeverityEnum.ERROR;
298         }
299     }

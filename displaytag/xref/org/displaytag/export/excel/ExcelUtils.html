135     /**
136      * We cache the styles; they are expensive to construct.
137      *
138      * @param properties
139      *            props for this run
140      */
141     public void initCellStyles(final TableProperties properties) {
142         // Integer
143         HSSFCellStyle style = this.getNewCellStyle();
144         style.setAlignment(HorizontalAlignment.RIGHT);
145         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_INTEGER)));
146         this.cellStyles.put(ExcelUtils.STYLE_INTEGER, style);
147 
148         // NUMBER
149         style = this.getNewCellStyle();
150         style.setAlignment(HorizontalAlignment.RIGHT);
151         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_NUMBER)));
152         this.cellStyles.put(ExcelUtils.STYLE_NUMBER, style);
153 
154         // Date
155         style = this.getNewCellStyle();
156         style.setAlignment(HorizontalAlignment.RIGHT);
157         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_DATE)));
158         style.setAlignment(HorizontalAlignment.RIGHT);
159         this.cellStyles.put(ExcelUtils.STYLE_DATE, style);
160 
161         // Long text
162         style = this.getNewCellStyle(); // http://jakarta.apache.org/poi/hssf/quick-guide.html#NewLinesInCells
163         style.setWrapText(true);
164         this.cellStyles.put(ExcelUtils.STYLE_LONGSTRING, style);
165 
166         // Regular text
167         this.cellStyles.put(ExcelUtils.STYLE_STRING, this.getNewCellStyle());
168 
169         this.wrapAt = Integer.parseInt(properties.getProperty(ExcelUtils.EXCEL_WRAPAT));
170     }

267     // patch from Karsten Voges
268     /**
269      * Escape certain values that are not permitted in excel cells.
270      *
271      * @param rawValue
272      *            the object value
273      *
274      * @return the escaped value
275      */
276     public static String escapeColumnValue(final Object rawValue) {
277         if (rawValue == null) {
278             return null;
279         }
280         String returnString = rawValue.toString();
281         // escape the String to get the tabs, returns, newline explicit as \t \r \n
282         returnString = StringEscapeUtils.escapeJava(StringUtils.trimToEmpty(returnString));
283         // remove tabs, insert four whitespaces instead
284         returnString = StringUtils.replace(StringUtils.trim(returnString), "\\t", "    ");
285         // remove the return, only newline valid in excel
286         returnString = StringUtils.replace(StringUtils.trim(returnString), "\\r", " ");
287         return StringEscapeUtils.unescapeJava(returnString);
288     }

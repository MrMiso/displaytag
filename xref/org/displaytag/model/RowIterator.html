22  package org.displaytag.model;
23  
24  import java.util.Iterator;
25  import java.util.List;
26  
27  import org.displaytag.decorator.TableDecorator;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /**
32   * Iterator on table rows.
33   *
34   * @author Fabrizio Giustina
35   *
36   * @version $Revision$ ($Author$)
37   */
38  public class RowIterator {
39  
40      /**
41       * logger.
42       */
43      private static Logger log = LoggerFactory.getLogger(RowIterator.class);
44  
45      /**
46       * internal iterator for Rows.
47       */
48      private final Iterator<Row> iterator;
49  
50      /**
51       * row number counter.
52       */
53      private int rowNumber;
54  
55      /**
56       * reference to the table TableDecorator.
57       */
58      private final TableDecorator decorator;
59  
60      /**
61       * id inherited from the TableTag (needed only for logging).
62       */
63      private String id;
64  
65      /**
66       * Starting offset for items n the current page. Needed to calculare the index in the original list
67       */
68      private final int pageOffset;
69  
70      /**
71       * Constructor for RowIterator.
72       *
73       * @param rowList
74       *            List containing Row objects
75       * @param columnList
76       *            List containing CellHeader objects
77       * @param tableDecorator
78       *            TableDecorator
79       * @param offset
80       *            Starting offset for items n the current page
81       */
82      protected RowIterator(final List<Row> rowList, final List<HeaderCell> columnList,
83              final TableDecorator tableDecorator, final int offset) {
84          this.iterator = rowList.iterator();
85          this.rowNumber = 0;
86          this.decorator = tableDecorator;
87          this.pageOffset = offset;
88      }
89  
90      /**
91       * Setter for the tablemodel id.
92       *
93       * @param tableId
94       *            same id of table tag, needed for logging
95       */
96      public void setId(final String tableId) {
97          this.id = tableId;
98      }
99  
100     /**
101      * Check if a next row exist.
102      *
103      * @return boolean true if a new row
104      */
105     public boolean hasNext() {
106         return this.iterator.hasNext();
107     }
108 
109     /**
110      * Gets the page offset.
111      *
112      * @return the page offset
113      */
114     public int getPageOffset() {
115         return this.pageOffset;
116     }
117 
118     /**
119      * Returns the next row object.
120      *
121      * @return Row
122      */
123     public Row next() {
124 
125         final int currentRowNumber = this.rowNumber++;
126 
127         if (RowIterator.log.isDebugEnabled()) {
128             RowIterator.log.debug("[{}] rowIterator.next() row number={}", this.id, currentRowNumber);
129         }
130 
131         final Row row = this.iterator.next();
132 
133         row.setRowNumber(currentRowNumber);
134 
135         if (this.decorator != null) {
136             this.decorator.initRow(row.getObject(), currentRowNumber, currentRowNumber + this.getPageOffset());
137         }
138 
139         return row;
140 
141     }
142 
143 }

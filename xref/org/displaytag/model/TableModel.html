434     /**
435      * Adds a column header (HeaderCell object).
436      *
437      * @param headerCell
438      *            HeaderCell
439      */
440     public void addColumnHeader(final HeaderCell headerCell) {
441         if (this.sortedColumnName == null) {
442             if (this.sortedColumn == this.headerCellList.size()) {
443                 headerCell.setAlreadySorted();
444             }
445         } else // the sorted parameter was a string so try and find that column name and set it as sorted
446         if (this.sortedColumnName.equals(headerCell.getSortName())) {
447             headerCell.setAlreadySorted();
448         }
449         headerCell.setColumnNumber(this.headerCellList.size());
450 
451         this.headerCellList.add(headerCell);
452     }

463     /**
464      * returns a RowIterator on the requested (full|page) list.
465      *
466      * @param full
467      *            if <code>true</code> returns an iterator on te full list, if <code>false</code> only on the viewable
468      *            part.
469      *
470      * @return RowIterator
471      *
472      * @see org.displaytag.model.RowIterator
473      */
474     public RowIterator getRowIterator(final boolean full) {
475         final RowIterator iterator = new RowIterator(full ? this.rowListFull : this.rowListPage, this.headerCellList,
476                 this.tableDecorator, this.pageOffset);
477         // copy id for logging
478         iterator.setId(this.id);
479         return iterator;
480     }

482     /**
483      * sorts the given list of Rows. The method is called internally by sortFullList() and sortPageList().
484      *
485      * @param list
486      *            List
487      */
488     private void sortRowList(final List<Row> list) {
489         if (this.isSorted()) {
490             final HeaderCell sortedHeaderCell = this.getSortedColumnHeader();
491 
492             // If it is an explicit value, then sort by that, otherwise sort by the property...
493             if ((sortedHeaderCell != null) && (sortedHeaderCell.getBeanPropertyName() != null
494                     || this.sortedColumn != -1 && this.sortedColumn < this.headerCellList.size())) {
495 
496                 final String sorted = sortedHeaderCell.getSortProperty() != null ? sortedHeaderCell.getSortProperty()
497                         : sortedHeaderCell.getBeanPropertyName();
498 
499                 Collections.sort(list, new RowSorter(this.sortedColumn, sorted, this.getTableDecorator(),
500                         this.sortOrderAscending, sortedHeaderCell.getComparator()));
501             }
502 
503         }
504 
505     }
506 
507     /**
508      * sort the list displayed in page.
509      */
510     public void sortPageList() {
511         if (TableModel.log.isDebugEnabled()) {
512             TableModel.log.debug("[{}] sorting page list", this.id);
513         }
514         this.sortRowList(this.rowListPage);
515 
516     }
517 
518     /**
519      * sort the full list of data.
520      */
521     public void sortFullList() {
522         if (TableModel.log.isDebugEnabled()) {
523             TableModel.log.debug("[{}] sorting full data", this.id);
524         }
525         this.sortRowList(this.rowListFull);
526     }

584     /**
585      * To string.
586      *
587      * @return the string
588      *
589      * @see java.lang.Object#toString()
590      */
591     @Override
592     public String toString() {
593         return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE) //
594                 .append("rowListFull", this.rowListFull) //$NON-NLS-1$
595                 .append("rowListPage", this.rowListPage) //$NON-NLS-1$
596                 .append("properties", this.properties) //$NON-NLS-1$
597                 .append("empty", this.isEmpty()) //$NON-NLS-1$
598                 .append("encoding", this.encoding) //$NON-NLS-1$
599                 .append("numberOfColumns", this.getNumberOfColumns()) //$NON-NLS-1$
600                 .append("headerCellList", this.headerCellList) //$NON-NLS-1$
601                 .append("sortFullTable", this.sortFullTable) //$NON-NLS-1$
602                 .append("sortedColumnNumber", this.getSortedColumnNumber()) //$NON-NLS-1$
603                 .append("sortOrderAscending", this.sortOrderAscending) //$NON-NLS-1$
604                 .append("sortedColumnHeader", this.getSortedColumnHeader()) //$NON-NLS-1$
605                 .append("sorted", this.isSorted()) //$NON-NLS-1$
606                 .append("tableDecorator", this.tableDecorator) //$NON-NLS-1$
607                 .append("caption", this.caption) // $NON-NLS-1
608                 .append("footer", this.footer) // $NON-NLS-1
609                 .append("media", this.media) // $NON-NLS-1
610                 .toString();
611     }

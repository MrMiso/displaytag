22  package org.displaytag.export.excel;
23  
24  import java.io.IOException;
25  import java.io.OutputStream;
26  
27  import jakarta.servlet.jsp.JspException;
28  
29  import org.apache.poi.hssf.usermodel.HSSFWorkbook;
30  import org.displaytag.model.HeaderCell;
31  import org.displaytag.render.HssfTableWriter;
32  
33  /**
34   * If you are doing an export, and do not know if the user would like a view that is grouped or a view that is raw, just
35   * use this one; it will export one tab that is grouped, and one tab that is not. Override the getHssfTableWriter method
36   * to change which HssfTableWriter is actually used.
37   *
38   * @author andy
39   *
40   * @see DefaultHssfExportView#getHssfTableWriter(org.apache.poi.hssf.usermodel.HSSFWorkbook) Date: Nov 13, 2010 Time:
41   *      8:46:29 AM
42   */
43  public class HssfDoubleExportView extends DefaultHssfExportView {
44  
45      /**
46       * Do export.
47       *
48       * @param out
49       *            the out
50       *
51       * @throws IOException
52       *             Signals that an I/O exception has occurred.
53       * @throws JspException
54       *             the jsp exception
55       */
56      @Override
57      public void doExport(final OutputStream out) throws IOException, JspException {
58          try {
59              final HSSFWorkbook wb = new HSSFWorkbook();
60              final HssfTableWriter writer = this.getHssfTableWriter(wb);
61              writer.setSetSheetName("Export");
62              writer.writeTable(this.model, "-1");
63  
64              boolean hasGroups = false;
65              for (final HeaderCell cell : this.model.getHeaderCellList()) {
66                  if (cell.getGroup() > 0) {
67                      hasGroups = true;
68                      cell.setGroup(0);
69                  }
70              }
71              if (hasGroups) {
72                  writer.setSetSheetName("Data");
73                  this.model.reset();
74                  writer.writeTable(this.model, "-1");
75              }
76              wb.write(out);
77              wb.close();
78          } catch (final Exception e) {
79              throw new HssfGenerationException(e);
80          }
81      }
82  }

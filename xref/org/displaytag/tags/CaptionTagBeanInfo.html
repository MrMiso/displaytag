22  package org.displaytag.tags;
23  
24  import java.beans.IntrospectionException;
25  import java.beans.PropertyDescriptor;
26  import java.beans.SimpleBeanInfo;
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  /**
31   * Needed to make the "class" tag attribute working.
32   *
33   * @author Fabrizio Giustina
34   *
35   * @version $Revision$ ($Author$)
36   */
37  public class CaptionTagBeanInfo extends SimpleBeanInfo {
38  
39      /**
40       * Gets the property descriptors.
41       *
42       * @return the property descriptors
43       *
44       * @see java.beans.BeanInfo#getPropertyDescriptors()
45       */
46      @Override
47      public PropertyDescriptor[] getPropertyDescriptors() {
48          final List<PropertyDescriptor> proplist = new ArrayList<>();
49  
50          try {
51              proplist.add(new PropertyDescriptor("class", //$NON-NLS-1$
52                      CaptionTag.class, null, "setClass")); //$NON-NLS-1$
53              proplist.add(new PropertyDescriptor("dir", //$NON-NLS-1$
54                      CaptionTag.class, null, "setDir")); //$NON-NLS-1$
55              proplist.add(new PropertyDescriptor("id", //$NON-NLS-1$
56                      CaptionTag.class, null, "setId")); //$NON-NLS-1$
57              proplist.add(new PropertyDescriptor("lang", //$NON-NLS-1$
58                      CaptionTag.class, null, "setLang")); //$NON-NLS-1$
59              proplist.add(new PropertyDescriptor("media", //$NON-NLS-1$
60                      CaptionTag.class, null, "setMedia")); //$NON-NLS-1$
61              proplist.add(new PropertyDescriptor("style", //$NON-NLS-1$
62                      CaptionTag.class, null, "setStyle")); //$NON-NLS-1$
63              proplist.add(new PropertyDescriptor("title", //$NON-NLS-1$
64                      CaptionTag.class, null, "setTitle")); //$NON-NLS-1$
65  
66              // make ATG Dynamo happy:
67              // Attribute "className" of tag "caption" in taglib descriptor file displaytag-11.tld" must have a
68              // corresponding property in class "org.displaytag.tags.CaptionTag" with a public setter method
69              proplist.add(new PropertyDescriptor("className", //$NON-NLS-1$
70                      CaptionTag.class, null, "setClass")); //$NON-NLS-1$
71          } catch (final IntrospectionException ex) {
72              throw new RuntimeException("You got an introspection exception - maybe defining a property that is not"
73                      + " defined in the CaptionTag?: " + ex.getMessage(), ex);
74          }
75  
76          final PropertyDescriptor[] result = new PropertyDescriptor[proplist.size()];
77          return proplist.toArray(result);
78      }
79  
80  }
